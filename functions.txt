import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';
const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type'
};
Deno.serve(async (req)=>{
  // CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', {
      headers: corsHeaders
    });
  }
  try {
    // 1. Validar variáveis de ambiente
    const supabaseUrl = Deno.env.get('SUPABASE_URL');
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
    if (!supabaseUrl || !supabaseServiceKey) {
      throw new Error('Missing environment variables');
    }
    // 2. Criar cliente com SERVICE ROLE
    const supabaseAdmin = createClient(supabaseUrl, supabaseServiceKey, {
      auth: {
        autoRefreshToken: false,
        persistSession: false
      }
    });
    // 3. Validar JWT do usuário
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) {
      return new Response(JSON.stringify({
        error: 'Unauthorized'
      }), {
        status: 401,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    const token = authHeader.replace('Bearer ', '');
    const { data: { user }, error: authError } = await supabaseAdmin.auth.getUser(token);
    if (authError || !user) {
      return new Response(JSON.stringify({
        error: 'Invalid token'
      }), {
        status: 401,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    // 4. Buscar lojas do usuário
    const { data: stores, error: storesError } = await supabaseAdmin.from('stores').select('id').eq('user_id', user.id);
    if (storesError) throw storesError;
    if (!stores || stores.length === 0) {
      return new Response(JSON.stringify({
        count: 0
      }), {
        status: 200,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    const storeIds = stores.map((s)=>s.id);
    // 5. Buscar itens pendentes
    const { data: items, error: itemsError } = await supabaseAdmin.from('order_items').select('order_id').in('store_id', storeIds).eq('status', 'pendente');
    if (itemsError) throw itemsError;
    // 6. Contar orders únicas
    const uniqueOrders = [
      ...new Set(items?.map((item)=>item.order_id) || [])
    ];
    return new Response(JSON.stringify({
      count: uniqueOrders.length
    }), {
      status: 200,
      headers: {
        ...corsHeaders,
        'Content-Type': 'application/json'
      }
    });
  } catch (error) {
    console.error('Error:', error);
    return new Response(JSON.stringify({
      error: error.message
    }), {
      status: 500,
      headers: {
        ...corsHeaders,
        'Content-Type': 'application/json'
      }
    });
  }
});



import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';
const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type'
};
Deno.serve(async (req)=>{
  // CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', {
      headers: corsHeaders
    });
  }
  try {
    // 1. Validar variáveis de ambiente
    const supabaseUrl = Deno.env.get('SUPABASE_URL');
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
    if (!supabaseUrl || !supabaseServiceKey) {
      throw new Error('Missing environment variables');
    }
    // 2. Criar cliente com SERVICE ROLE (bypassa RLS)
    const supabaseAdmin = createClient(supabaseUrl, supabaseServiceKey, {
      auth: {
        autoRefreshToken: false,
        persistSession: false
      }
    });
    // 3. Validar JWT do usuário
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) {
      return new Response(JSON.stringify({
        error: 'Unauthorized'
      }), {
        status: 401,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    const token = authHeader.replace('Bearer ', '');
    const { data: { user }, error: authError } = await supabaseAdmin.auth.getUser(token);
    if (authError || !user) {
      return new Response(JSON.stringify({
        error: 'Invalid token'
      }), {
        status: 401,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    // 4. Validar parâmetro 'view'
    const { view } = await req.json();
    if (view && ![
      'vendedor',
      'comprador'
    ].includes(view)) {
      return new Response(JSON.stringify({
        error: 'Invalid view parameter'
      }), {
        status: 400,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    // ========================================
    // VISÃO VENDEDOR
    // ========================================
    if (view === 'vendedor') {
      // Buscar lojas do usuário
      const { data: stores, error: storesError } = await supabaseAdmin.from('stores').select('id').eq('user_id', user.id);
      if (storesError) throw storesError;
      if (!stores || stores.length === 0) {
        return new Response(JSON.stringify({
          data: []
        }), {
          status: 200,
          headers: {
            ...corsHeaders,
            'Content-Type': 'application/json'
          }
        });
      }
      const storeIds = stores.map((s)=>s.id);
      // Buscar order_items das lojas do vendedor
      const { data: items, error: itemsError } = await supabaseAdmin.from('order_items').select(`
          *,
          product_listings (
            products (
              images
            )
          )
        `).in('store_id', storeIds).order('created_at', {
        ascending: false
      });
      if (itemsError) throw itemsError;
      if (!items || items.length === 0) {
        return new Response(JSON.stringify({
          data: []
        }), {
          status: 200,
          headers: {
            ...corsHeaders,
            'Content-Type': 'application/json'
          }
        });
      }
      // Buscar orders relacionadas
      const orderIds = [
        ...new Set(items.map((item)=>item.order_id))
      ];
      const { data: orders, error: ordersError } = await supabaseAdmin.from('orders').select('*').in('id', orderIds).order('created_at', {
        ascending: false
      });
      if (ordersError) throw ordersError;
      // Montar resposta (apenas itens do vendedor)
      const pedidos = orders?.map((order)=>({
          ...order,
          order_items: items.filter((item)=>item.order_id === order.id)
        })) || [];
      return new Response(JSON.stringify({
        data: pedidos
      }), {
        status: 200,
        headers: {
          ...corsHeaders,
          'Content-Type': 'application/json'
        }
      });
    }
    // ========================================
    // VISÃO COMPRADOR (padrão)
    // ========================================
    const { data: orders, error: ordersError } = await supabaseAdmin.from('orders').select(`
        *,
        order_items (
          *,
          product_listings (
            products (
              images
            )
          )
        )
      `).eq('user_id', user.id).order('created_at', {
      ascending: false
    });
    if (ordersError) throw ordersError;
    return new Response(JSON.stringify({
      data: orders || []
    }), {
      status: 200,
      headers: {
        ...corsHeaders,
        'Content-Type': 'application/json'
      }
    });
  } catch (error) {
    console.error('Error:', error);
    return new Response(JSON.stringify({
      error: error.message
    }), {
      status: 500,
      headers: {
        ...corsHeaders,
        'Content-Type': 'application/json'
      }
    });
  }
});
